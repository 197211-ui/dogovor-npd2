<script>
    (function() {
        // Правильные ответы: для каждого вопроса индекс правильного варианта (0 или 1)
        const correctAnswers = [1, 0, 1, 0, 1, 0];
        const selects = document.querySelectorAll('.option-select');
        const checkBtn = document.getElementById('checkBtn');
        const resetBtn = document.getElementById('resetBtn');
        const resultMsg = document.getElementById('resultMessage');

        function resetAll() {
            selects.forEach(sel => {
                sel.value = '';
                sel.classList.remove('correct', 'incorrect');
                sel.disabled = false;
            });
            resultMsg.textContent = 'выберите все варианты';
            resultMsg.className = 'result-message';
        }

        function checkAll() {
            selects.forEach(sel => {
                sel.classList.remove('correct', 'incorrect');
            });

            let allSelected = true;
            let correctCount = 0;
            const total = selects.length;

            selects.forEach((sel, index) => {
                const val = sel.value;
                if (val === '') {
                    allSelected = false;
                    return;
                }
                const selectedNum = parseInt(val, 10);
                const isCorrect = (selectedNum === correctAnswers[index]);
                if (isCorrect) {
                    sel.classList.add('correct');
                    correctCount++;
                } else {
                    sel.classList.add('incorrect');
                }
                sel.disabled = true;
            });

            if (!allSelected) {
                resultMsg.textContent = '⚠️ выберите вариант для каждого пункта';
                resultMsg.className = 'result-message error';
                selects.forEach(sel => {
                    if (sel.value !== '') {
                        sel.disabled = true;
                    } else {
                        sel.disabled = false;
                    }
                });
                selects.forEach(sel => {
                    if (sel.value === '') {
                        sel.classList.remove('correct', 'incorrect');
                    }
                });
                return;
            }

            if (correctCount === total) {
                resultMsg.textContent = '✅ Все формулировки безопасны!';
                resultMsg.className = 'result-message success';
            } else {
                resultMsg.textContent = `❌ ${total - correctCount} пунктов требуют корректировки.`;
                resultMsg.className = 'result-message error';
            }

            selects.forEach(sel => { sel.disabled = true; });
        }

        checkBtn.addEventListener('click', checkAll);

        resetBtn.addEventListener('click', function() {
            resetAll();
        });

        selects.forEach(sel => {
            sel.addEventListener('change', function() {
                if (this.disabled) {
                    this.disabled = false;
                    this.classList.remove('correct', 'incorrect');
                    resultMsg.textContent = 'выберите все варианты';
                    resultMsg.className = 'result-message';
                } else {
                    this.classList.remove('correct', 'incorrect');
                }
            });
        });

        resetAll();
    })();
</script>
